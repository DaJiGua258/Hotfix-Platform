-- 玩家控制器
require("FSM/StateMachine")
require("FSM/IdleState")
require("FSM/RunState")
require("FSM/JumpState")
require("FSM/FallState")
require("FSM/DashState")

Object:subClass("PlayerCtrl")

-- 触发处理（全局函数，由 PlayerMotor.OnTriggerEnter 调用）
function OnPlayerTrigger(obj)

    if obj.tag == "Flag" then
        SceneData:SaveData()
        GameOver:Show()
        GamePanel:SetVCamActive(false)
        CS.UnityEngine.Time.timeScale = 0
        isGameOver = true

    elseif obj.tag == "Coin" then
        SceneData:AddCoin()
        GamePanel:AddCoin()
        obj:SetActive(false)
        _G.AudioMgr:PlayCoin()
    end
end

function PlayerCtrl:new()
    local obj = Object.new(self)

    -- 从game包加载Player预制体，生成在起点位置
    local playerObj = ABMgr:LoadRes("game", "Player", typeof(GameObject))
    local startPoint = GameObject.FindGameObjectWithTag("Start")
    if startPoint then
        local pos = startPoint.transform.position
        playerObj.transform.position = Vector3(pos.x, pos.y + 3, pos.z)
        obj.spawnPos = Vector3(pos.x, pos.y + 3, pos.z)
    end
    obj.motor = playerObj:GetComponent(typeof(CS.PlayerMotor))
    obj.input = playerObj:GetComponent(typeof(CS.InputProxy))

    -- 清除场景中已有摄像机，从game包加载Camera并设置跟随玩家
    local oldCam = GameObject.FindGameObjectWithTag("MainCamera")
    if oldCam then
        CS.UnityEngine.GameObject.Destroy(oldCam)
    end
    -- 加载相机
    local cameraObj = ABMgr:LoadRes("game", "Camera", typeof(GameObject))
    local vcam = cameraObj:GetComponentInChildren(typeof(CS.Cinemachine.CinemachineVirtualCamera))
    if vcam then
        vcam.Follow = playerObj.transform
        vcam.LookAt = playerObj.transform
    end

    obj.stateMachine = StateMachine:new()  -- 状态机

    -- 渐入渐出状态
    obj.fadeState = nil

    -- 二段跳计数
    obj.jumpCount = 0

    -- 状态类
    obj.idleState = IdleState:new()
    obj.runState = RunState:new()
    obj.fallState = FallState:new()
    obj.jumpState = JumpState:new()
    obj.dashState = DashState:new()

    obj.stateMachine:ChangeState(obj.idleState, obj)  -- 初始化状态机
    return obj
end

function PlayerCtrl:Update()
    if isGameOver then return end

    -- 掉落检测，渐入渐出后传送
    if self.motor.transform.position.y < -8 and self.spawnPos then
        if not self.fadeState then
            self.fadeState = true
            Fade:FadeIn(0.2, function()
                self.motor:Teleport(self.spawnPos)
                Fade:FadeOut(0.2, function()
                    self.fadeState = nil
                end)
            end)
        end
    end

    self.stateMachine:Update(self)
end

return PlayerCtrl
