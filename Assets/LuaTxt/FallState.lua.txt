require("FSM/State")

State:subClass("FallState")

function FallState:OnEnter(ctrl)
end

function FallState:OnUpdate(ctrl)
    -- 空中突进
    if ctrl.input.IsDashPressed then
        ctrl.stateMachine:ChangeState(ctrl.dashState, ctrl)
        return
    end

    -- 二段跳
    if ctrl.input.IsJumpPressed and ctrl.jumpCount < 1 then
        ctrl.motor:AirJump()
        ctrl.jumpCount = ctrl.jumpCount + 1
        ctrl.stateMachine:ChangeState(ctrl.jumpState, ctrl)
        return
    end

    if ctrl.motor.IsGrounded then
        ctrl.stateMachine:ChangeState(ctrl.idleState, ctrl)
    end
end

function FallState:OnExit(ctrl)
    _G.AudioMgr:PlayLand()
end

return FallState
