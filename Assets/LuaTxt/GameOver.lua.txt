BasePanel:subClass("GameOver")

function GameOver:Init(name)
    self.base.Init(self, name)

    -- 缓存下一关按钮
    self.nextLevelBtn = self:GetComp("btn_NextLevel", "Button")
    self.nextLevelGo = self:GetComp("btn_NextLevel", "Button").gameObject

    self:GetComp("btn_ReStart", "Button").onClick:AddListener(
        function()
            CS.UnityEngine.Time.timeScale = 1
            isGameOver = false
            LoadSceneFromAB("level", CS.UnityEngine.SceneManagement.SceneManager.GetActiveScene().name)
        end
    )

    self:GetComp("btn_ReturnMain", "Button").onClick:AddListener(
        function()
            CS.UnityEngine.Time.timeScale = 1
            isGameOver = false
            CS.UnityEngine.SceneManagement.SceneManager.LoadScene("Main")
        end
    )

    self.nextLevelBtn.onClick:AddListener(
        function()
            CS.UnityEngine.Time.timeScale = 1
            isGameOver = false
            LoadSceneFromAB("level", SceneData:GetNextLevel())
        end
    )
end

function GameOver:Show()
    self.base.Show(self, "panel_GameOver")
    CS.UnityEngine.Cursor.visible = true

    -- 显示通关时间和金币
    local minutes = math.floor(CurLevelData.time / 60)
    local seconds = math.floor(CurLevelData.time % 60)
    local timeStr = string.format("完成时间：%02d分%02d秒", minutes, seconds)
    local coinStr = "金币收集：" .. CurLevelData.coin
    self.panelObj.transform:Find("Content/LevelTitle/text_Score"):GetComponent(typeof(Text)).text = timeStr .. "\n" .. coinStr

    -- 没有下一关时隐藏按钮
    self.nextLevelGo:SetActive(SceneData:GetNextLevel() ~= nil)

    -- 加载星星纹理
    local star = ABMgr:LoadRes("ui", "star", typeof(Sprite))
    local star_outline = ABMgr:LoadRes("ui", "star_outline", typeof(Sprite))

    -- 根据分数替换星星纹理
    local scoreRoot = self.panelObj.transform:Find("Content/Score")
    local scoreImgs = scoreRoot:GetComponentsInChildren(typeof(Image))
    local score = CurLevelData.score
    for i = 0, scoreImgs.Length - 1 do
        if score > 0 then
            scoreImgs[i].sprite = star
            score = score - 1
        else
            scoreImgs[i].sprite = star_outline
        end
    end
end

function GameOver:Hide()
    self.base.Hide(self)
end
